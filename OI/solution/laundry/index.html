8.21 T3 laundry

\(T3\ \ laundry\)

该题主要运用贪心思路：

洗衣服用时最大的和烘干衣服用时最小的结合，方案最优

可以建两个优先队列，队内从小到大排序，一次正序取，另一次倒序取

由于我结构体用了一半，懒得改了，所有的结构体都可以换成 pair ，但可读性较差

第 6 行等同于 

#define ll long long

第 11 ~ 14 行，第 17 ~ 21 行，使用 pair 则可以省去

第 52 ~ 55 行较难理解，其意思是：对于第 \(i\) 台洗衣机，洗第 \(k\) 件衣服要花 \(w_i \times k\) 的时间，因为一台洗衣机不能同时洗两件衣服，所以第 \(k\) 件衣服还需等待前 \(k - 1\) 件衣服洗完，即等待 \(w_i \times (k - 1)\) 的时间

#include <iostream>
#include <cstdio>
#include <queue>
#include <algorithm>
using namespace std;
typedef long long ll;

struct mach
{
    ll now_time; int id;//洗下一件衣服需要的时间（w[i] + 等待时间），编号 
    mach(ll x, int y)
    {
        now_time = x, id = y;
    }//构造函数 
};

bool operator > (mach x, mach y)
{
    return x.now_time > y.now_time; 
} //重载运算符，用于优先队列
//当然，你也可以用 pair 

priority_queue<mach, vector<mach>, greater<mach> > wash, dry;
//洗衣机，烘干机，从小到大 

ll n, m, l, w[(int)1e7+10], d[(int)1e6+10], _time[(int)1e6+10], ans;
//_time 存储每件衣服洗完 + 烘干 + 等待所用的时间 
int main()
{
    freopen("laundry.in","r",stdin);
    freopen("laundry.out","w",stdout);

    cin >> l >> n >> m;
    for (int i = 1; i <= n; i++)
    {
        cin >> w[i]; 
        wash.push(mach(w[i], i));
    }
    for (int i = 1; i <= m; i++)
    {
        cin >> d[i];
        dry.push(mach(d[i], i));
    }
    //输入，按从小到大排列 

    for (int i = 1; i <= l; i++)//从第一件到第 l 件衣服 
    {
        ll tmp_time = wash.top().now_time;
        int tmp_id = wash.top().id;
        wash.pop();
        //取出队首衣服 
        _time[i] = tmp_time;
        //第 i 件衣服洗完 + 等待的时间 
        //该洗衣机洗下一件衣服需要等待该件衣服洗完，故加上洗该件衣服所需时间
        wash.push(mach(tmp_time + w[tmp_id], tmp_id));
    }
    for(int i = l; i >= 1; i--)//从 l 件衣服到第 1 件衣服
    //这样就保证了洗衣服用时最小的和烘干衣服用时最大的结合 
    {
        ll tmp_time = dry.top().now_time;
        int tmp_id = dry.top().id;
        dry.pop();
        //取出队首衣服 
        _time[i] += tmp_time;
        //第 i 件衣服洗完 + 烘干 + 等待的时间 
        //该烘干机烘干下一件衣服需要等待该件衣服烘干完，故加上烘干该件衣服所需时间 
        dry.push(mach(tmp_time + d[tmp_id], tmp_id));
        ans = max(ans, _time[i]);//最大的即为答案 
    }
    cout << ans;

    fclose(stdin);
    fclose(stdout);
    return 0;
}
